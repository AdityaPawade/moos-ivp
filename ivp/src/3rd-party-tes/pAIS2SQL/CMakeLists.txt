FILE(GLOB SRC *.cpp)

OPTION(IVP_BUILD_pAIS2SQL "Build program pAIS2SQL" ON)

IF("${IVP_BUILD_pAIS2SQL}" STREQUAL "ON")
   ADD_EXECUTABLE(pAIS2SQL ${SRC})

   INCLUDE_DIRECTORIES(/usr/include/mysql)
   
   TARGET_LINK_LIBRARIES(pAIS2SQL 
   mysqlclient
   MOOS
   MOOSGen
   MOOSUtility
   mbutil
   m
   pthread)

   IF("${IVP_PACKAGE_VERSION}" STREQUAL "")
      MESSAGE(STATUS "pAIS2SQL: IVP_PACKAGE_VERSION not set, so not creating install component.")
   ELSE("${IVP_PACKAGE_VERSION}" STREQUAL "")
      INSTALL(
         TARGETS pAIS2SQL
         DESTINATION bin/ivp-${IVP_PACKAGE_VERSION}
         COMPONENT none)
   ENDIF("${IVP_PACKAGE_VERSION}" STREQUAL "")

ENDIF("${IVP_BUILD_pAIS2SQL}" STREQUAL "ON")
