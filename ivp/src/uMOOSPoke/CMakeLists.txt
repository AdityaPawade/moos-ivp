FILE(GLOB SRC *.cpp)

OPTION(IVP_BUILD_uMOOSPoke "Build program uMOOSPoke" OFF)

IF("${IVP_BUILD_uMOOSPoke}" STREQUAL "ON")
   ADD_EXECUTABLE(uMOOSPoke ${SRC})
   
   TARGET_LINK_LIBRARIES(uMOOSPoke 
   MOOS
   MOOSGen
   m
   pthread)

   IF("${IVP_PACKAGE_VERSION}" STREQUAL "")
      MESSAGE(STATUS "uMOOSPoke: IVP_PACKAGE_VERSION not set, so not creating install component.")
   ELSE("${IVP_PACKAGE_VERSION}" STREQUAL "")
      INSTALL(
         TARGETS uMOOSPoke
         DESTINATION bin/ivp-${IVP_PACKAGE_VERSION}
         COMPONENT ivp-tools)
   ENDIF("${IVP_PACKAGE_VERSION}" STREQUAL "")

ENDIF("${IVP_BUILD_uMOOSPoke}" STREQUAL "ON")
