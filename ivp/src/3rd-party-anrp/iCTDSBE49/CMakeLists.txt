FILE(GLOB SRC *.cpp)

OPTION(IVP_BUILD_iCTDSBE49 "Build program iCTDSBE49" OFF)

IF("${IVP_BUILD_iCTDSBE49}" STREQUAL "ON")
   ADD_EXECUTABLE(iCTDSBE49 ${SRC})
   
   TARGET_LINK_LIBRARIES(iCTDSBE49 
   MOOS
   MOOSGen
   anrp_util
   m
   pthread)

   IF("${IVP_PACKAGE_VERSION}" STREQUAL "")
      MESSAGE(STATUS "iCTDSBE49: IVP_PACKAGE_VERSION not set, so not creating install component.")
   ELSE("${IVP_PACKAGE_VERSION}" STREQUAL "")
      INSTALL(
         TARGETS iCTDSBE49
         DESTINATION bin/ivp-${IVP_PACKAGE_VERSION}
         COMPONENT ivp-tools)
   ENDIF("${IVP_PACKAGE_VERSION}" STREQUAL "")

ENDIF("${IVP_BUILD_iCTDSBE49}" STREQUAL "ON")
